Introduction

    src="/++resource++multiple_file_upload-1.48/jquery.MultiFile.js">&#160;</script>

    src="/++resource++gsfileupload-20121206.js">&#160;</script>

    jQuery.noConflict();

        GSFileUpload.init();

  </script>
